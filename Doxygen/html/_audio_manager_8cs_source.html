177 public void PlaySoundEffect(SoundEffect soundEffect, float volume = 1.0f)
178 {
179 AudioClip clip = PickRandomAudioClip(_SoundEffects[soundEffect]);
180 bool played = false;
181
182 for (int i = 0; i < _soundEffectSources.Count; i++)
183 {//look for an avaialable channel
184 if (!_soundEffectSources[i].isPlaying)
185 {//if its free play the sound effect
186 _soundEffectSources[i].pitch = Random.Range(0.8f, 1.2f);
187 _soundEffectSources[i].PlayOneShot(clip, volume);
188 played = true;
189 break;
190 }
191 }
192
193 if (!played)
194 {
195 Debug.LogWarning("Not played Sound Effect: " +
196 soundEffect.ToString() +
197 " no audio source available");
